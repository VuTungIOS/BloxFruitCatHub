-- Hàm Auto Farm
function AutoFarmFunc()
    spawn(function()
        while getgenv().AutoFarm do
            task.wait(0.3)

            local player = game.Players.LocalPlayer
            local char = player.Character
            local root = char and char:FindFirstChild("HumanoidRootPart")
            if not root then continue end

            local nearest, dist = nil, math.huge
            for _, mob in pairs(workspace:GetChildren()) do
                if mob:FindFirstChild("Humanoid") 
                and mob:FindFirstChild("HumanoidRootPart") 
                and mob.Name:lower():find("zombie") then
                    local mag = (mob.HumanoidRootPart.Position - root.Position).Magnitude
                    if mag < dist then
                        nearest, dist = mob, mag
                    end
                end
            end

            if nearest and nearest:FindFirstChild("Humanoid") and nearest.Humanoid.Health > 0 then
                root.CFrame = nearest.HumanoidRootPart.CFrame * CFrame.new(0, 0, -3)
                pcall(function()
                    local tool = char:FindFirstChildOfClass("Tool")
                    if tool then
                        tool:Activate()
                    end
                end)
            end
        end
    end)
end

-- Hàm Auto Pick Item
function AutoPickFunc()
    spawn(function()
        while getgenv().AutoPick do
            task.wait(0.2)

            local player = game.Players.LocalPlayer
            local char = player.Character
            local root = char and char:FindFirstChild("HumanoidRootPart")
            if not root then continue end

            for _, item in pairs(workspace:GetChildren()) do
                if item:IsA("Tool") or item:FindFirstChild("TouchInterest") or item:FindFirstChildWhichIsA("ProximityPrompt") then
                    pcall(function()
                        if item:IsA("Model") and item:FindFirstChild("PrimaryPart") then
                            root.CFrame = item.PrimaryPart.CFrame + Vector3.new(0, 2, 0)
                        elseif item:IsA("BasePart") then
                            root.CFrame = item.CFrame + Vector3.new(0, 2, 0)
                        end
                        firetouchinterest(root, item, 0)
                        firetouchinterest(root, item, 1)
                    end)
                end
            end
        end
    end)
end
